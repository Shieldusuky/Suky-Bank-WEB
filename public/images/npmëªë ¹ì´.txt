[Mysql DB 순번 초기화 명령어]

1) 문법 - id 순서 초기화
ALTER TABLE [테이블명] AUTO_INCREMENT=1;
SET @COUNT = 0;
UPDATE [테이블명] SET [컬럼명] = @COUNT:=@COUNT+1;

2) 예문
ALTER TABLE users AUTO_INCREMENT=1;
SET @COUNT = 0;
UPDATE users SET id = @COUNT:=@COUNT+1;

ALTER TABLE beneficiaries AUTO_INCREMENT=1;
SET @COUNT = 0;
UPDATE beneficiaries SET id = @COUNT:=@COUNT+1;

ALTER TABLE transactions AUTO_INCREMENT=1;
SET @COUNT = 0;
UPDATE transactions SET id = @COUNT:=@COUNT+1;
